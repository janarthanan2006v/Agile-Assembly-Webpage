<script>
    // --- Page Management Elements ---
    const homePage = document.getElementById('home-page');
    const dashboardPage = document.getElementById('dashboard-page');
    const developersPage = document.getElementById('developers-page');
    const featuresPage = document.getElementById('features-page');
    const infoPage = document.getElementById('info-page');
    const projectsPage = document.getElementById('projects-page'); // New Projects Page element

    
    // --- Function to switch between pages ---
    function showPage(pageId) {
        // Hide all page sections first
        document.querySelectorAll('.page-section').forEach(section => {
            section.classList.remove('active');
            // Remove 'visible' class from all elements when switching pages
            section.querySelectorAll('.animate-on-scroll').forEach(el => el.classList.remove('visible'));
            // Reset about text slides when leaving home page
            if (section.id === 'infoSection') {
                section.querySelectorAll('.about-text-slide').forEach(textEl => textEl.classList.remove('slide-in'));
            }
            // Reset service description slides when leaving home page
            if (section.id === 'services-section') {
                section.querySelectorAll('.service-description-text').forEach(textEl => textEl.classList.remove('slide-in'));
            }
        });

        // Show the requested page section
        let targetPage;
        if (pageId === 'home') {
            targetPage = homePage;
        } else if (pageId === 'dashboard') {
            targetPage = dashboardPage;
        } else if (pageId === 'developers') {
            targetPage = developersPage;
        } else if (pageId === 'features') {
            targetPage = featuresPage;
        } else if (pageId === 'info') {
            targetPage = infoPage;
        } else if (pageId === 'projects') {
            targetPage = projectsPage;
        }

        if (targetPage) {
            targetPage.classList.add('active');
            // Manually trigger animation for elements within the newly active page
            // This ensures they animate even if not scrolled into view (e.g., if the page is short)
            targetPage.querySelectorAll('.animate-on-scroll').forEach(el => {
                // Use a small timeout to allow display:block to apply before animation
                setTimeout(() => {
                    el.classList.add('visible');
                }, 50);
            });

            // Trigger about text slideshow if it's the infoSection on the home page
            if (targetPage.id === 'home-page') { // Check if the home-page is active
                const aboutTextSlides = document.querySelectorAll('#infoSection .about-text-slide');
                aboutTextSlides.forEach((textEl, index) => {
                    setTimeout(() => {
                        textEl.classList.add('slide-in');
                    }, 500 * (index + 1)); // Delay each sentence
                });

                // Trigger service description slideshow if it's the services-section on the home page
                const serviceDescriptionTexts = document.querySelectorAll('#services-section .service-description-text');
                serviceDescriptionTexts.forEach((textEl, index) => {
                    setTimeout(() => {
                        textEl.classList.add('slide-in');
                    }, 500 * (index + 1)); // Delay each description
                });
            }


            // Scroll to the top of the newly activated page
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
    }

    // --- Header Shadow on Scroll ---
    const fixedHeaderContainer = document.getElementById('fixedHeaderContainer');
    window.addEventListener('scroll', () => {
        if (window.scrollY > 0) {
            fixedHeaderContainer.classList.add('header-shadow');
        } else {
            fixedHeaderContainer.classList.remove('header-shadow');
        }
    });

    // IntersectionObserver to trigger animations on scroll
const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
        if (entry.isIntersecting) {
            entry.target.classList.add('visible'); // Add the 'visible' class to trigger animation
        }
    });
}, {
    threshold: 0.5  // Trigger when 50% of the element is in view
});

// Observe all elements with the 'animate-on-scroll' class
document.querySelectorAll('.animate-on-scroll').forEach(element => {
    observer.observe(element);
});


    // --- Event Listeners for Navigation Links ---
    document.querySelectorAll('.main-nav-links a').forEach(link => {
        link.addEventListener('click', (e) => {
            e.preventDefault();
            const targetId = e.target.getAttribute('href').substring(1); // Get ID without '#'

            if (targetId === 'home-page') {
                showPage('home');
            } else if (targetId === 'info-page') {
                showPage('info');
            } else if (targetId === 'features-page') {
                showPage('features');
            }
            else if (targetId === 'dashboard-page') {
                showPage('dashboard');
            } else if (targetId === 'developers-page') {
                showPage('developers');
            } else if (targetId === 'projects-page') {
                showPage('projects');
            }
        });
    });

    // Initialize the page view on load
    document.addEventListener('DOMContentLoaded', () => {
        showPage('home'); // Always start on the home page
    });

</script>
